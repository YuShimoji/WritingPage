2. 「TT」要素を特定

// ツールバー内のすべてのテキストノードを検索
function findTextNodes(element) {
  const nodes = [];
  const walk = document.createTreeWalker(element, NodeFilter.SHOW_TEXT, null, false);
  let node;
  while (node = walk.nextNode()) {
    if (node.textContent.trim() === 'TT' || node.textContent.includes('TT')) {
      nodes.push({
        text: node.textContent,
        parent: node.parentElement,
        parentClass: node.parentElement.className,
        parentId: node.parentElement.id
      });
    }
  }
  return nodes;
}

const toolbar = document.querySelector('.toolbar');
const ttNodes = findTextNodes(toolbar || document.body);
console.log('Found TT nodes:', ttNodes);

3. ツールバーの全ボタンを表示

// ツールバーの全ボタンを表示
const buttons = document.querySelectorAll('.toolbar button, .toolbar-actions button');
buttons.forEach((btn, i) => {
  console.log(`Button ${i}:`, {
    id: btn.id,
    class: btn.className,
    text: btn.textContent.trim().substring(0, 50),
    visible: window.getComputedStyle(btn).display !== 'none'
  });
});
